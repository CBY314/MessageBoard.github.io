                    <h4>修改管理员账号名</h4>
                    <input type="text" id="newAdminName" placeholder="输入新管理员名称">
                    <button onclick="changeAdminName()">确认修改</button>
                </div>
                <div style="margin-top:15px;">
                    <input type="text" id="newU" placeholder="新增账号">
                    <input type="password" id="newP" placeholder="密码">
                    <button onclick="addUser()">添加账号</button>
                </div>
                <div style="margin-top:10px;">
                    <input type="text" id="editU" placeholder="操作账号">
                    <input type="password" id="editP" placeholder="新密码">
                    <button onclick="modifyUser()">修改密码</button>
                    <button onclick="removeUser()" class="admin-del">删除账号</button>
                </div>
                <div id="userListShow" style="margin-top:15px;"></div>
            </div>

            <div class="input-box">

    <script>
        let userList = [];
        let msgList = [];
        let isAdmin = false;
        let nowLoginUser = "";

        window.onload = function(){
            initData();
            loadMsg();
        }

        function initData(){
            let old = localStorage.getItem("userList");
            if(old){
                userList = JSON.parse(old);
            }else{
                userList = [{
                    name: "CBY",
                    pwd: "CBY888123",
                    admin: true
                }];
                localStorage.setItem("userList", JSON.stringify(userList));
            }
        }

        function saveUser(){
            localStorage.setItem("userList", JSON.stringify(userList));
        }

        function login(){
            let u = document.getElementById("inpUser").value.trim();
            let p = document.getElementById("inpPwd").value.trim();
            for(let i=0;i<userList.length;i++){
                if(userList[i].name === u && userList[i].pwd === p){
                    isAdmin = userList[i].admin;
                    nowLoginUser = u;
                    goHome();
                    return;
                }
            }
            alert("账号或密码错误！");
        }

        function goHome(){
            document.getElementById("loginBox").style.display = "none";
            document.getElementById("wallBox").style.display = "block";
            if(isAdmin){
                document.getElementById("adminPanel").style.display = "block";
                renderUserList();
            }else{
                document.getElementById("adminPanel").style.display = "none";
            }
            renderMsg();
        }

        function logout(){
            location.reload();
        }

        function changeAdminName(){
            let newName = document.getElementById("newAdminName").value.trim();
            if(!newName){alert("名称不能为空");return;}
            if(userList.some(x => x.name === newName)){alert("账号名已被占用");return;}
            let admin = userList.find(x => x.admin === true);
            if(admin){
                admin.name = newName;
                nowLoginUser = newName;
                saveUser();
                alert("修改成功，请重新登录");
                logout();
            }
        }

        function addUser(){
            let u = document.getElementById("newU").value.trim();
            let p = document.getElementById("newP").value.trim();
            if(!u||!p){alert("账号密码不能为空");return;}
            if(userList.some(x=>x.name===u)){alert("账号已存在");return;}
            userList.push({name:u,pwd:p,admin:false});
            saveUser();
            alert("添加成功，可以直接登录");
            document.getElementById("newU").value="";
            document.getElementById("newP").value="";
            renderUserList();
        }

        function modifyUser(){
            let u = document.getElementById("editU").value.trim();
            let p = document.getElementById("editP").value.trim();
            let item = userList.find(x=>x.name===u);
            if(!item){alert("账号不存在");return;}
            if(item.admin){alert("不能修改管理员");return;}
            item.pwd = p;
            saveUser();
            alert("修改成功");
            renderUserList();
        }

        function removeUser(){
            let u = document.getElementById("editU").value.trim();
            let item = userList.find(x=>x.name===u);
            if(!item){alert("账号不存在");return;}
            if(item.admin){alert("不能删除管理员");return;}
            if(!confirm("确定删除？"))return;
            userList = userList.filter(x=>x.name!==u);
            saveUser();
            alert("删除成功");
            renderUserList();
        }

        function renderUserList(){
            let html = "<h4>📋 所有账号（含密码）</h4>";
            userList.forEach(i=>{
                html+=`<div class="user-item">
                    <span>账号：${i.name} ${i.admin?"【管理员】":""}</span>
                    <span>密码：${i.pwd}</span>
                </div>`;
            });
            document.getElementById("userListShow").innerHTML=html;
        }

        function getTime(){
            let d=new Date();
            return `${d.getFullYear()}-${d.getMonth()+1}-${d.getDate()} ${d.getHours()}:${d.getMinutes()}`;
        }

        function loadMsg(){
            let m=localStorage.getItem("msgList");
            if(m)msgList=JSON.parse(m);
        }
        function saveMsg(){
            localStorage.setItem("msgList",JSON.stringify(msgList));
        }

        function sendMsg(){
            let t=document.getElementById("msgTxt").value.trim();
            if(!t){alert("留言不能为空");return;}
            msgList.push({user:nowLoginUser,time:getTime(),content:t,isTop:false,topOrder:0});
            saveMsg();
            renderMsg();
            document.getElementById("msgTxt").value="";
        }

        function toggleTop(idx){
            msgList[idx].isTop = !msgList[idx].isTop;
            if(msgList[idx].isTop){
                let topArr = msgList.filter(m=>m.isTop);
                msgList[idx].topOrder = topArr.length;
            }else{
                msgList[idx].topOrder = 0;
            }
            saveMsg();
            renderMsg();
        }

        function moveUp(idx){
            if(!msgList[idx].isTop) return;
            let currOrder = msgList[idx].topOrder;
            if(currOrder <= 1) return;
            for(let i=0;i<msgList.length;i++){
                if(msgList[i].isTop && msgList[i].topOrder === currOrder - 1){
                    let temp = msgList[i].topOrder;
                    msgList[i].topOrder = msgList[idx].topOrder;
                    msgList[idx].topOrder = temp;
                    saveMsg();
                    renderMsg();
                    return;
                }
            }
        }

        function moveDown(idx){
            if(!msgList[idx].isTop) return;
            let currOrder = msgList[idx].topOrder;
            let topArr = msgList.filter(m=>m.isTop);
            if(currOrder >= topArr.length) return;
            for(let i=0;i<msgList.length;i++){
                if(msgList[i].isTop && msgList[i].topOrder === currOrder + 1){
                    let temp = msgList[i].topOrder;
                    msgList[i].topOrder = msgList[idx].topOrder;
                    msgList[idx].topOrder = temp;
                    saveMsg();
                    renderMsg();
                    return;
                }
            }
        }

        function delMsg(i){
            if(!confirm("确定删除？"))return;
            msgList.splice(i,1);
            saveMsg();
            renderMsg();
        }

        function renderMsg(){
            let b=document.getElementById("msgList");
            b.innerHTML="";
            let topMsgs = msgList.filter(m=>m.isTop).sort((a,b)=>a.topOrder - b.topOrder);
            let normalMsgs = msgList.filter(m=>!m.isTop).reverse();

            topMsgs.forEach((item)=>{
                let idx = msgList.findIndex(m=>m.user===item.user && m.time===item.time && m.content===item.content);
                renderOneMsg(idx,true);
            });
            normalMsgs.forEach((item)=>{
                let idx = msgList.findIndex(m=>m.user===item.user && m.time===item.time && m.content===item.content);
                renderOneMsg(idx,false);
            });
        }

        function renderOneMsg(idx,isTop){
            let i2 = msgList[idx];
            let d=document.createElement("div");
            d.className="msg-item" + (isTop ? " msg-top" : "");

            let btnHtml = "";
            if(isAdmin){
                btnHtml += `<button class="top-btn" onclick="toggleTop(${idx})">${isTop?'取消置顶':'置顶'}</button>`;
                if(isTop){
                    btnHtml += `<button class="move-btn" onclick="moveUp(${idx})">上移</button>`;
                    btnHtml += `<button class="move-btn" onclick="moveDown(${idx})">下移</button>`;
                }
                btnHtml += `<button class="del-msg-btn" onclick="delMsg(${idx})">删除</button>`;
            }else{
                if(i2.user === nowLoginUser){
                    btnHtml += `<button class="del-msg-btn" onclick="delMsg(${idx})">删除</button>`;
                }
            }

            d.innerHTML=`
                <div class="msg-info">${isTop?'📌 置顶｜':''}发布人：${i2.user}｜时间：${i2.time}</div>
                <div class="msg-content">${i2.content}</div>
                ${btnHtml}
            `;
            document.getElementById("msgList").appendChild(d);
        }
    </script>
